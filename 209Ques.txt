Create a Class Main which would be used to accept an integer and call the static method present in UserMainCode.
Note: First two numbers in a Fibonacci series are 0, 1 and all other subsequent numbers are sum of its previous two numbers. Example - 0, 1, 1, 2, 3, 5...
Input and Output Format:
Input consists of an integer, which corresponds to n.
Output consists of an Integer (sum of fibonacci numbers).
Refer sample output for formatting specifications.
Sample Input :
5
Sample Output :
7
[0 + 1 + 1 + 2 + 3 = 7]
public class UserMainCode {
            public static int getSumOfNFibos(int n) {
                        int first = 0, second = 1, next = 0, sum = 0;
                        ArrayList<Integer> list = new ArrayList<Integer>();
                        for (int i = 0; i < n; i++) {
                                    if (i == 0) {
                                                next = first;
                                    }
                                    if (i == 1) {
                                                next = second;
                                    }
                                    if (i > 1) {
                                                next = second + first;
                                                first = second;
                                                second = next;
                                    }
                                    list.add(next);
                        }
                        Iterator<Integer> iterator = list.iterator();
                        while (iterator.hasNext()) {
                                    Integer integer = iterator.next();
                                    sum = sum + integer;
                        }
                        return sum;
            }
}

A U B ={ 7,11,12,14,24,26,36} and

                        ArrayList<Integer> al = new ArrayList<Integer>(max);

                        for (int i = 0; i < a.length; i++) {
                                    if (!al.contains(a[i])) {
                                                ++count;
                                                al.add(a[i]);

                                    if (!al.contains(b[i])) {
                                                ++count;
                                                al.add(b[i]);
                                    }
                        }
                        union = new int[count];
                        for (int i = 0; i < count; i++) {
                                    union[i] = (int) (al.get(i));
                        }
                        Arrays.sort(union);
 
                        // Intersection
 
                        al = new ArrayList<Integer>(max);
                        count = 0;

                                                ++count;

                                    inter[i] = (int) (al.get(i));
